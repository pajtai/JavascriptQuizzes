Quiz 1

Consider the code at the bottom of the page...

6 rows will be displayed. One for each execution of the methods returned by the closure.
  • How many times will "obj.count" be undefined?
  • How many times wil the value of "count" and "value" be equal on a single line?
  • Will the values for "count" be the same for runs 1&4, 2&5, and 3&6?
  • Will the values for "obj.count" be the same for runs 1&4, 2&5, and 3&6?
    (function(){
    var i = 1,        
        closure = function(valueIn, objIn) {
            var  count = i,
                 value = valueIn,                 
                 obj   = objIn;
            return function() {
                return "count is: "     + count      + " " +
                       "value is: "     + value      + " " +
                       "obj.count is: " + obj.count  + "<br/>";
            };   
        };
    $(function() {
        var one, two, three, theObject = {};
        
        one   = closure(i++, theObject);
        two   = closure(++i, theObject);
        three = closure(i + 1, theObject);    
        
        $("body").append("1: " + one());
        $("body").append("2: " + two());
        $("body").append("3: " + three());
        theObject.count = 5;
        $("body").append("4: " + one());
        $("body").append("5: " + two());
        $("body").append("6: " + three());
    });
}());
