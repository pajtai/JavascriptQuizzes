Quiz 2:

Consider the code at the bottom of the page... (uses jQuery in places not relating to the question)

3 rows will be displayed. Two for "getAnswer()" and one for "getOtherAnswer()"
  • State the name of what causes the first answer.
  • What will the 3 answers be?
  • Hoisting
(function() {
    var answer = 42,
        getAnswer = function() {
            $("body").append("The answer is: " + answer + "<br/>");
            
            var answer = answer;
            $("body").append("The answer is: " + answer + "<br/>");           
        },
        getOtherAnswer = function() {            
            $("body").append("The answer is: " + answer + "<br/>");
        };
    
    $(function() {
        getAnswer();
        getOtherAnswer();
    });
}());
