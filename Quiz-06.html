Quiz 6:

Consider the code at the bottom of the page... (uses jQuery in places not relating to the question)

The program will print three lines.
  1. What one word will lines one and two share?
  2. What is the word line three will display?
$(function() {
    var one, two, i;
    
    Object.prototype.hello = "hi ";
    Object.prototype.talk  = "interesting ";
    Object.prototype.bye   = "ciao ";
    
    Function.prototype.hello = "hello ";
    Function.prototype.talk  = "okay ";
    Function.prototype.bye   = "adios ";

    one = function() { return this; };  
    
    delete Function.prototype.hello;
    
    two = function(varIn) { return this[varIn]; };
    one.hello = "g'day ";
    
    $("body").append(one.hello + one.talk + one().bye    + "<br/>");
    $("body").append(two.hello + two.talk + two("hello") + "<br/>");
    
    for(i in one) {
        if (one.hasOwnProperty(i))
            $("body").append(i);        
    }
});
